{{ message }}

  el.div({ id: 'app-5' }, [
    el.p('{{  message  }}'),
    el.button({ 'v-on:click': 'reverseMessage' }, 'Reverse Message')
  ]),
  ex.script(() => {
    window.app5 = new Vue({
      el: '#app-5',
      data: {
        message: 'Hello Vue.js!'
      },
      methods: {
        reverseMessage: function () {
          this.message = this.message
            .split('')
            .reverse()
            .join('')
        }
      }
    })
  }),
